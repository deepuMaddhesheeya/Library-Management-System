---create tables
---Books Tables
CREATE TABLE Books
(
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Title NVARCHAR(150) NOT NULL,
    Author NVARCHAR(150) NOT NULL,
    Genre NVARCHAR(100),
    IsIssued BIT DEFAULT 0
);

---Members Table

CREATE TABLE Members
(
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(150) NOT NULL,
    Email NVARCHAR(150)
);
